import tkinter as tk

# 创建主窗口
root = tk.Tk()
root.geometry('430x200')

# 创建按钮
button = tk.Button(root, text="Button", width=34, height=2)

# 将按钮居中
button.grid(row=0, column=0, padx=(root.winfo_width()-button.winfo_reqwidth())//2, pady=(root.winfo_height()-button.winfo_reqheight())//2)
root.grid_rowconfigure(0, weight=1)
root.grid_columnconfigure(0, weight=1)

# 进入事件循环
root.mainloop()
